Leslie matrix example

# Create your matrix ----
insect_leslie <- matrix(data = c(0, 0, 600, 0.2, 0, 0, 0, 0.08, 0), 
                        nrow = 3, 
                        ncol = 3,
                        byrow = TRUE)
insect_leslie
     [,1] [,2] [,3]
[1,]  0.0 0.00  600
[2,]  0.2 0.00    0
[3,]  0.0 0.08    0
# initial starting conditions ----

insect_y0 <- c(12000, 700, 500)
insect_y0
[1] 12000   700   500
# Populations at year 1 ----
insect_y1 <- insect_leslie %*% insect_y0
insect_y1
       [,1]
[1,] 300000
[2,]   2400
[3,]     56
# populations at year 2 ----
insect_y2 <- insect_leslie %*% insect_y1

# Population at year 3 ----
insect_y3 <- insect_leslie %*% insect_y2

insect_y3
       [,1]
[1,] 115200
[2,]   6720
[3,]   4800

Another Leslie matrix example - Trees

# make a matrix ----
tree_leslie <- matrix(c(0, 0, 5000, 0.1, 0, 0, 0, 0.06, 0.95), nrow = 3, ncol = 3, byrow = TRUE)

tree_leslie
     [,1] [,2]    [,3]
[1,]  0.0 0.00 5000.00
[2,]  0.1 0.00    0.00
[3,]  0.0 0.06    0.95
# initial starting conditions ----
tree_y0 <- c(5000, 0, 0)

# pop at year 1 ----
tree_y1 <- tree_leslie %*% tree_y0
tree_y1
     [,1]
[1,]    0
[2,]  500
[3,]    0
# pop at year 2 ----
tree_y2 <- tree_leslie %*% tree_y1
tree_y2
     [,1]
[1,]    0
[2,]    0
[3,]   30
